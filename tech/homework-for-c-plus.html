        m_length = length;
        m_width = width;

    {

    }

void converByAddress(struct CartesianCoordinates * coordinate, struct PolarCoordinate polar);
void converByReference(struct CartesianCoordinates & coordinate, struct PolarCoordinate polar);
struct PolarCoordinate
{
    double radius;
    double phaseAngle;
};
struct CartesianCoordinates
{
    double x;
    double y;
};

int main(int argc, char const * argv[])
{
    //Passing by address.
    struct CartesianCoordinates crtsByAddress, crtsByReference;
    struct PolarCoordinate polar;
    cout << "Input the radius of polar coordinates : ";
    cin >> polar.radius;
    cout << "Input the phase angle of polar coordinates : ";
    cin >> polar.phaseAngle;
    converByAddress(&crtsByAddress, polar);
    cout << "The value computed by passing address : ( " << crtsByAddress.x << "," << crtsByAddress.y << ").\n"; 
    converByReference(crtsByReference, polar);
    cout << "The value computed by passing address : ( " << crtsByReference.x << "," << crtsByReference.y << ").\n"; 
    return 0;
}
void converByAddress(struct CartesianCoordinates * coordinate, struct PolarCoordinate polar)
{
    coordinate->x = cos(polar.radius) * polar.phaseAngle;
    coordinate->y = sin(polar.radius) * polar.phaseAngle;
}
void converByReference(struct CartesianCoordinates & coordinate, struct PolarCoordinate polar)
{
    coordinate.x = cos(polar.radius) * polar.phaseAngle;
    coordinate.y = sin(polar.radius) * polar.phaseAngle;
}
